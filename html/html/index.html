window.addEventListener('load',function(){
    const canvas = document.querySelector('canvas')
    const ctx = canvas.getContext('2d')

    canvas.height = window.innerHeight
    canvas.width = window.innerWidth

    let painting = false

    canvas.addEventListener('mousedown',startPosition)
    canvas.addEventListener('mouseup',finishedPosition)
    canvas.addEventListener('mousemove',draw)

    function startPosition(e){
        painting = true
        draw(e)
    }
    function finishedPosition(e){
        painting = false
        ctx.beginPath()
    }
    function draw(e){
        if(!painting) return
        // ctx.beginPath()
        ctx.lineWidth = 10
        ctx.lineCap = 'round'

        ctx.lineTo(e.clientX,e.clientY)
        ctx.stroke()
    }   
})
</script>

<canvas id="canvas" width="400" height="400"></canvas>
<script>
    const canvas = document.querySelector('#canvas')
    const ctx = canvas.getContext('2d')
    const circle = {
        x:200,
        y:200,
        r:30,
        dx:5,
        dy:4
    }

    function drawCircle(){
        ctx.beginPath()
        ctx.arc(circle.x,circle.y,circle.r,0,Math.PI*2)
        ctx.fillStyle = 'pink'
        ctx.fill()
    }

    function update(){
        //在指定的矩形区域内清除像素，x,y矩形的起始点
        ctx.clearRect(0,0,canvas.width,canvas.height)
        drawCircle()
        circle.x += circle.dx
        circle.y += circle.dy
        //detect side walls
        if(circle.x + circle.r > canvas.width ||circle.x - circle.r < 0){
            circle.dx = circle.dx * -1
        }
        if(circle.y + circle.r > canvas.height || circle.y - circle.r < 0){
            circle.dy = circle.dy * -1
        }
        window.requestAnimationFrame(update)
    }
    update()
</script>
</body>
</html>

<script>
const image = document.querySelector('#source')
const canvas = document.getElementById("canvas");
canvas.width = window.innerWidth;
canvas.height = window.innerHeight;
const ctx = canvas.getContext("2d");
const player = {
    w: 50,
    h: 70,
    x: 20,
    y: 200,
    speed: 5,
    dx: 0,
    dy: 0
}

function drawPlayer(){
    ctx.drawImage(image,player.x,player.y,player.w,player.h)
}
function clear(){
        ctx.clearRect(0,0,canvas.width,canvas.height)
}
function newPos(){
    player.x += player.dx
    player.y += player.dy
    detectWalls()
}
function detectWalls(){
    if(player.x < 0) player.x = 0
    if(player.x + player.w > canvas.width) player.x = canvas.width - player.w
    if(player.y < 0) player.y = 0
    if(player.y + player.h > canvas.height) player.y = canvas.height - player.h
}
function update(){
    clear()
    drawPlayer()
    newPos()
    requestAnimationFrame(update)
}

function moveUp(){
    player.dy =  - player.speed
}
function moveDown(){
    player.dy =  player.speed
}
function moveLeft(){
    player.dx =  - player.speed
}
function moveRight(){
    player.dx =  player.speed
}
function keyDown(e){
    if(e.key === 'ArrowRight' || e.key === 'Right'){
        moveRight()
    } else if(e.key === 'ArrowLeft' || e.key === 'Left'){
        moveLeft()
    } else if(e.key === 'ArrowUp' || e.key === 'Up'){
        moveUp()
    } else if(e.key === 'ArrowDown' || e.key === 'Down'){
        moveDown()
    }
}
function keyUp(e){
    console.log(1)

        e.key == 'Right' ||

        e.key == 'Left' ||
        e.key == 'ArrowLeft' ||
        e.key == 'Up' ||
        e.key == 'ArrowUp'||
        e.key == 'Down' ||
        e.key == 'ArrowDown'
    ){
        player.dx = 0
        player.dy = 0
    }
}
update()

    Canvas是什么¶

    • <canvas>是HTML的一个标签。
    • <canvas>标签用于创建一个画布。
      • 画布是矩形的
      • 画布没有边框
      • 画布没有内容
      • 画布仅仅是一块画布
    • <canvas>画布上可以绘图，但是只能使用js绘图。

    定义画布¶

    必须指定id，width,height属性。可以使用css增加边框。 

    <canvas id="myCanvas" width="200" height="100" style="border:1px solid #000000;">
</canvas>

    添加JS¶

    创建完canvas画布后，只能使用js画画。

    <!--1.指定id以供js使用 2.默认尺寸：300*150  -->
    <canvas id="myCanvas"></canvas>


    <script>
        //2.获取canvas元素
        const canvas = document.querySelector('#myCanvas') 
        //3.引入画布的绘画环境类型：2d(CanvasRenderingContext2D)|webgl
        const ctx = canvas.getContext('2d')
        //4.设置canvas宽度
        canvas.width = window.innerWidth
        canvas.height = window.innerHeight
        //6.
        ctx.fillStyle = 'red'
        //5. fillRect(x,y,width,height)绘制带有填充色的矩形。默认黑色。
        //x:矩形左上角
        ctx.fillRect(20,20,150,100)
        ctx.fillStyle = 'blue'
        ctx.fillRect(200,20,150,100)

        //7. strokeRect()
        ctx.lineWidth = '5px'
        ctx.fillStyle = 'green'
        ctx.strokeRect(100,200,150,100)

        //clearRect()
        ctx.clearRect(25,25,140,90)

        //fillText()
        ctx.font = '30px arial'
        ctx.fillStyle = 'orange'
        ctx.fillText('Hello world!', 400,50)
        //strokeText()
        ctx.font = '30px arial'
        ctx.strokeStyle = 'red'
        ctx.strokeText('Hello world!', 400,100)
        //绘制四边形
        ctx.beginPath()//开始一条路径，重置一条路径
        ctx.moveTo(350,200)//移动路径到指定点
        ctx.lineTo(550,200)//创建一条线，增加一个新点
        ctx.lineTo(450,400)//创建一条线，增加一个新点
        ctx.closePath()
        ctx.stroke()
        ctx.fill()
        //绘制直线三角形
        ctx.beginPath()//开始一条路径，重置一条路径
        ctx.moveTo(450,400)//移动路径到指定点
        ctx.lineTo(550,200)//创建一条线，增加一个新点
        ctx.lineTo(650,400)//创建一条线，增加一个新点
        ctx.closePath()
        ctx.stroke()
        ctx.fill()
        //绘制矩形
        ctx.beginPath()
        ctx.rect(300,50,150,100)//创建矩形：x坐标 y坐标 宽度 高度
        ctx.fill()

        //arc()
        const centerX = canvas.width / 2
        const centerY = canvas.height / 2
        ctx.beginPath()
        ctx.arc(centerX,centerY,300,0,Math.PI*2,false)//创建弧线：x,y,r,sAngle,eAngle,counterclockwise
        ctx.moveTo(centerX + 200,centerY)
        ctx.arc(centerX,centerY,200,0,Math.PI,false)
        ctx.moveTo(centerX - 130,centerY - 150)
        ctx.arc(centerX - 150,centerY - 150,20,0,Math.PI*2)
        ctx.moveTo(centerX + 170,centerY - 150)
        ctx.arc(centerX + 150,centerY - 150,20,0,Math.PI*2)
        ctx.stroke()


    </script>
