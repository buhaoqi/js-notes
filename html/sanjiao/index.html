    let W = canvas.width = 800;
    let H = canvas.height = 600;

    let balls = [];

    for(let i=0; i<10; i++){
      balls.push(new Ball({
        x: Math.random() * W,
        y: Math.random() * H,
        r: Math.random() * 20 + 30,
        fillStyle: `#${Math.floor(Math.random()*16777215).toString(16)}`,
        vx: (Math.random() - 0.5) * 3,
        vy: (Math.random() - 0.5) * 3,
        id: `ball:` + i
      }));
    }

    function ballMove(ball, index){
      ball.x += ball.vx;
      ball.y += ball.vy;

      if(ball.x - ball.r >= W || ball.x + ball.r <= 0 ||
      ball.y - ball.r >= H || ball.y + ball.r <= 0){
        balls.splice(index, 1);
        if(balls.length){
          console.log(`${ball.id}这个小球被移除了!`);
        }else{
          console.log(`所有的小球都被移除了!`);
        }
      }

      ball.render(ctx);
    }

    (function move(){
      window.requestAnimationFrame(move);

      ctx.clearRect(0, 0, W, H);

      let i = balls.length;

      while(i--){
        ballMove(balls[i], i);
      }
    })();

const canvas = document.getElementById('canvas');
const ctx = canvas.getContext('2d');

let W = canvas.width = 800;
let H = canvas.height = 600;

let vx = 0, vy = 0, vr = 0, a = 0, w = 46, h = 20, f = 0.65;

const arrow = new Arrow({
    x: W/2,
    y: H/2,
    w,
    h
}).render(ctx);

window.addEventListener('keydown', function (e){
    console.log(e.keyCode)
    switch(e.keyCode){
    case 37:

        vr = -5;
        break;
    case 39:
        vr = 5;
        break;
    case 38:
        a = 0.5;
        break;
    }
});

window.addEventListener('keyup', function (e){
    a = 0;
    vr = 0;
});

(function move(){
    window.requestAnimationFrame(move);

    ctx.clearRect(0, 0, W, H);

    arrow.rotation += vr * Math.PI/180
    let angle = arrow.rotation;
    let ax = Math.cos(angle) * a;
    let ay = Math.sin(angle) * a;

    vx += ax;
    vy += ay;

    vx *= f;
    vy *= f;

    arrow.x += vx;
    arrow.y += vy;

    // 越界处理
    if(arrow.x - arrow.w / 2 >= W){
    arrow.x = 0 - arrow.w / 2;
    }else if(arrow.x + arrow.w/2 <= 0){
    arrow.x = W + arrow.w/2;
    }

    arrow.render(ctx);
})();

</script>
</body>

const canvas = document.getElementById('canvas');
const ctx = canvas.getContext('2d');

let W = canvas.width = 800;
let H = canvas.height = 600;

class Ball2 extends Ball {
    constructor(opt){
        super()
        this.vy = 0
        this.g = 0
        this.friction = 0.05
        this.firstMove = true
        Object.assign(this, opt)
        return this
    }
    move(){

        if(!this.firstMove){
            this.vy += this.g
        }

        if(this.vy > 0 && this.vy - this.friction > 0){
            this.vy -= this.friction
        } else if(this.vy < 0 && this.vy + this.friction < 0){
            this.vy += this.friction
        } else {
            this.vy = 0
        }
        this.y += this.vy
        if( this.y >= H - this.r){
            this.y = H - this.r
            this.vy *= -0.8
        }
        this.firstMove = false
        return this
    }
}

const balls = []
for(let i = 0; i < 100; i++){
    const ball2 = new Ball2({
        x: Math.random() * W,
        y: Math.random() * (H/2),
        g: Math.random() * 0.2 + 0.1 ,
        r: Math.random() * 5 + 5,
        fillStyle: `#${Math.floor(Math.random()*16777215).toString(16)}`
    })
    balls.push(ball2)
}
console.log(balls)
;(function falling(){
    let id = window.requestAnimationFrame(falling)
    ctx.clearRect(0, 0, W, H)
    balls.forEach((ball, i) => ball.move().render(ctx))
})()

//create an instance of the Ball()
const ball = new Ball({
    x: W/2,

//store position(x,y)
let step = 1
let direction = 1
;(function round(){
    //start animation
    window.requestAnimationFrame(round)

    ctx.clearRect(0, 0, W, H)

    ball.x = W/10 + step
    //key1
    if(step >= W/10 * 8) direction = -1
    if(step <= 0 ) direction = 1
    if (direction == -1) step -= 3
    if (direction == 1) step += 3
    //create ball instance
    ball.render(ctx)

const ball = new Ball({
    x: W/2,
    y: H/2,
    r: 30
}).render(ctx)

//variables
let rx = 200, ry = 80
let angle = 0
let speed = 0.02

;(function round(){
    window.requestAnimationFrame(round)
    ctx.clearRect(0, 0, W, H)
    ctx.save()
    ctx.translate(W/2, H/2)
    //canvas并不提供椭圆的api，手动设置椭圆

    ctx.arc(0, 0, rx, 0, Math.PI * 2, false)
    ctx.stroke()
    ctx.restore()

    ball.x = W/2 + rx * Math.cos(angle)
    ball.y = H/2 + ry * Math.sin(angle)

    angle += speed
    angle %= Math.PI * 2

    ball.render(ctx)

    let angle = Math.atan2(dy,dx)

    let vy = speed * Math.sin(angle)

    arrow.y += vy
    console.log(arrow.x,arrow.y)

//create canvas context
const canvas = document.querySelector('#canvas')
const ctx = canvas.getContext('2d')
let W = canvas.width = 600
let H = canvas.height = 400

//create an instance of the Arrow()
const ball = new Ball({
    x: W/2,
    y: H/10, 
    r: 30
}).render(ctx)

let g = 0.2, vy = 0
;(function move(){
    window.requestAnimationFrame(move)
    ctx.clearRect(0, 0, W, H)
    //位置+速度的变化
    ball.y += vy
    //变化+加速度（加速度就是改变物体速度的力量）
    vy += g
    if(ball.y + ball.r >= H){
        ball.y = H - ball.r
        vy *= -0.8
    }

    ball.render(ctx)
})()
