处理滑屏误触：自定义tab事件
思路：在手指抬起时有没有发生move事件
如果没发生 认定用户要点击*/
function Tap(el,fn){
  let isMove = false
  el.addEventListener('touchmove',function(ev){
      isMove = true
  })
  el.addEventListener('touchend',function(ev){
      if(!isMove){
          fn&&fn()
      }
      isMove = false
  })
}

    UIEvent接口表示用户界面事件。UIEvent继承自Event，它主要包括以下几个子集： - MouseEvent - TouchEvent - FocusEvent - KeyboardEvent - WheelEvent - InputEvent - CompositionEvent

    Touch Events¶

    定义： - TouchEvent 接口表示一个 UIEvent（用户界面事件）。 - 触发条件：当**触摸点**与**触感表面**的**接触状态**发生变化时触发该事件。 - 触摸点：表面上的一个接触点。通常是手指，也可以是身体的某一个部分，或是触摸笔。 - 触感表面：可感知触摸的平面/表面，可以是屏幕或触控板。 - 接触状态：开始、结束、在表面移动 - 功能： - 触摸事件提供了在触摸屏或触控板上解释手指（或触控笔）活动的能力。 - 触摸事件为程序提供了多点触控交互的支持。 - 触摸事件被描述为一个或多个点。包括支持检测运动、添加删除接触点等。 - 多点触控交互的整个交互期间，程序接收开始、移动、结束三个阶段的触摸事件： - 开始：当一个手指接触屏幕时 - 随后：其他手指也可触摸设备表面并随意进行滑动 - 结束：当所有手指离开时结束 - 于MouseEvent的不同之处 - TouchEvent提供同一表面不同位置的同步触摸 - TouchEvent接口将当前所有活动的触摸点封装起来。 - Touch 接口表示单独一个触摸点，其中包含参考浏览器视角的相对坐标。

    触摸对象表示多个触摸点。每一个触摸点描述为一个位置，尺寸和形状，压力，目标元素， 触摸点列表通过TouchList对象表示。

    以下事件不存在 - swip - rotate - pinch - zoom - tap - double tap

        }

    for(var i = 1; i < navs.length; i++){
      var rect = navs[i].getBoundingClientRect();
      if(rect.top <= pointY && rect.bottom >= pointY){
        setNav(navs[i]);
        toScroll(navs[i]);

    for(var i = 1; i < navs.length; i++){
      navs[i].className = "";
    }
    index.className = "focus";
  }
  /* 找到选中的导航找到对应的dt */
  function getOption(index){
    var dataId = index.dataset.index;
    //console.log(dataId);

   // console.log(option);

      index = index.nextElementSibling;
      if(!index){
        //如果下一项不存在，说明已经到了列表的最后
        option = indexList.querySelectorAll('dt');
        option = option[option.length-1];//那当前对应的就是列表的最后一项
      } else {
         option = getOption(index);//如果下一项存在 就去找它下一项对应的选项
      }
    }
    return option;
  }

  /* 根据找到的dt，计算出滚动条位置 */
  function toScroll(index){/* 滚动至dt所在的位置 */
     var option = getOption(index);
     var scrollY = getOffsetY(option);//获取dt在页面中的位置
     window.scrollTo(0,scrollY);
  }

  /* 获取元素在页面中的位置*/
  function getOffsetY(el){
    var scrollY = window.scrollY;
    var rectY = el.getBoundingClientRect().top;
    return scrollY + rectY;
  }

        if(!next||getOffsetY(next) > pageY){

<script type="text/javascript">
function createList(){
    var list = document.querySelector('#list');
    var inner = "";
    for(var i = 0; i < 100; i++){
        inner += "<li>这是第"+i+"个li</li>"
    }   
    list.innerHTML = inner;
}
document.addEventListener('touchstart', function(e) {
    e.preventDefault();
});
window.onload = function(){
    createList();
    var wrap = document.querySelector('#wrap');
    var list = document.querySelector('#list');
    var startPoint = 0;
    var startEl = 0;
    var lastY = 0;
    var lastDis = 0;
    var lastTime = 0;
    var lastTimeDis = 0;
    var maxTranslate = wrap.clientHeight - list.offsetHeight;
    css(list,"translateZ",0.01);
    list.addEventListener('touchstart', function(e) {
        maxTranslate = wrap.clientHeight - list.offsetHeight;
        clearInterval(list.timer);
        startPoint = e.changedTouches[0].pageY;
        startEl = css(list,"translateY");
        lastY = startEl;
        lastTime = new Date().getTime();
        lastTimeDis = lastDis = 0;
    });
    list.addEventListener('touchmove', function(e) {
        var nowTime = new Date().getTime();
        var nowPoint = e.changedTouches[0].pageY;
        var dis = nowPoint - startPoint;
        var translateY = startEl + dis;
        css(list,"translateY",translateY);
        lastDis = translateY - lastY;
        lastY = translateY;
        lastTimeDis = nowTime - lastTime;
        lastTime = nowTime;
    });
    list.addEventListener('touchend', function(e) {
        var type = "easeOut";
        var speed = Math.round(lastDis / lastTimeDis*10);
        speed = lastTimeDis <= 0?0 :speed;
        var target = Math.round(speed*30 + css(list,"translateY"));
        if(target > 0){
            target = 0;
            type = "backOut";
        } else if(target < maxTranslate){
            target = maxTranslate;
            type = "backOut";
        }
        console.log(target);
        MTween({
            el:list,
            target: {translateY:target},
            time: Math.round(Math.abs(target - css(list,"translateY"))*2),
            type: type
        });
    });
};
</script>
