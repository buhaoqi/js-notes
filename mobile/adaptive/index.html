    mySlider()
function mySlider(){
const sliderClient = document.querySelector('#slider1')
const sliderContainer = document.querySelector('.slider-box')
const dots = sliderClient.querySelectorAll('span')
let n1 = 0
let n2 = 0
let imgWidth = sliderClient.offsetWidth
let timer = null
let iStartTouchX = 0 //记录StartTouchX
let iStartX = 0 //记录当前StartX
autoPlay()
function autoPlay(){
    timer = setInterval(function(){
        n1++
        n1 = n1%dots.length
        console.log(n1,dots.length)
        slider()
    },2000) 
}
sliderClient.addEventListener('touchstart',function(ev){
    clearInterval(timer)
    // alert(1)
    ev = ev.changedTouches[0]// 手指列表，只记录一个
    iStartTouchX = ev.pageX
    iStartX = n2
})
// document.addEventListener('touchmove',function(ev){
//     ev.preventDefault()
// })
sliderClient.addEventListener('touchmove',function(ev){
    sliderContainer.style.transition = 'none'
    ev = ev.changedTouches[0]
    let iDis = ev.pageX - iStartTouchX
    n2 = iStartX + iDis
    sliderContainer.style.transform = 'translateX('+ n2 + 'px)' 
})
sliderClient.addEventListener('touchend',function(ev){
    n1 = n2 / imgWidth
    n1 = -Math.round(n1)
    if(n1<0){
        n1 = 0
    }
    if(n1>dots.length - 1){
        n1 = dots.length - 1
    }
    slider()
    autoPlay()
})

function slider(){
    n2 = -n1 * imgWidth
    console.log(n1,n2)
    sliderContainer.style.transition = '0.5s'
    sliderContainer.style.transform = 'translateX('+ n2 + 'px)'
    for(var i=0;i<dots.length;i++){
        dots[i].classList.remove('active')
    }
    dots[n1].classList.add('active')
}
}
